    private void Awake()
    {
       // DynamicMap.onMapChanged += HandleMapChanged;
     //   DynamicMap.onShowTypesChanged += HandleShowTypesChanged;
        Instance = this;
        _harmony = new Harmony("com.debug.eventlogger");
        _harmony.PatchAll();
        Debug.Log("[DebugEventLogger] Awake: patches applied.");
        StartCoroutine(WaitAndSubscribeInstanceEvents());
    }

    private void OnDestroy()
    {
        UnsubscribeFromAircraft();
        _harmony.UnpatchSelf();
      //  DynamicMap.onMapChanged -= HandleMapChanged;
     //   DynamicMap.onShowTypesChanged -= HandleShowTypesChanged;
        if (DynamicMap.i != null)
        {
            var dm = DynamicMap.i;
        //    dm.onMapGenerated -= HandleMapGenerated;
            dm.onMapMaximized -= HandleMapMaximized;
            dm.onMapMinimized -= HandleMapMinimized;
        //    dm.onAllDeselected -= HandleAllDeselected;
       //     dm.onUnitSelected -= HandleUnitSelected;
      //      dm.onUnitDeselected -= HandleUnitDeselected;
        }
        Debug.Log("[DebugEventLogger] OnDestroy: patches removed.");
    }

    private void SubscribeToAircraft(Aircraft a)
    {
        UnsubscribeFromAircraft();          // clean up old sub

        if (a == null) return;              // nothing to hook

        _subscribedAircraft = a;

        // ------------- aircraft-level events -------------
        a.onSpawnedInPosition += OnSpawned;
        a.onEject += OnEject;
        a.OnTouchdown += OnTouchdown;
        a.onSortieSuccessful += OnSortieSuccessful;
        a.onSetGear += OnSetGear;
        a.onSetFlightAssist += OnFlightAssist;
        //   a.onShake += OnShake;
        //a.onRadarWarning += OnRadarWarning;
        a.onInitialize += OnUnitInitialize;
        a.onDisableUnit += OnUnitDisable;
        a.onChangeFaction += OnUnitChangeFaction;
        a.onRearm += OnUnitRearm;
        a.onRepair += OnUnitRepair;
        a.onAddIRSource += OnUnitAddIRSource;
        a.onRegisterMissile += OnUnitRegisterMissile;
        a.onDeregisterMissile += OnUnitDeregisterMissile;
        a.onJam += OnUnitJam;

        foreach (var ws in a.weaponStations)
            HookWeaponStation(ws);

        HookMissileWarning(a.GetMissileWarningSystem());

        Debug.Log($"[DebugEventLogger] Subscribed to Aircraft '{a.name}' (+ MissileWarning)");
    }

    private void UnsubscribeFromAircraft()
    {
        if (_subscribedAircraft != null)
        {
            var a = _subscribedAircraft;

            a.onSpawnedInPosition -= OnSpawned;
            a.onEject -= OnEject;
            a.OnTouchdown -= OnTouchdown;
            a.onSortieSuccessful -= OnSortieSuccessful;
            a.onSetGear -= OnSetGear;
            a.onSetFlightAssist -= OnFlightAssist;
            //   a.onShake -= OnShake; 
            //a.onRadarWarning -= OnRadarWarning;
            a.onInitialize -= OnUnitInitialize;
            a.onDisableUnit -= OnUnitDisable;
            a.onChangeFaction -= OnUnitChangeFaction;
            a.onRearm -= OnUnitRearm;
            a.onRepair -= OnUnitRepair;
            a.onAddIRSource -= OnUnitAddIRSource;
            a.onRegisterMissile -= OnUnitRegisterMissile;
            a.onDeregisterMissile -= OnUnitDeregisterMissile;
            a.onJam -= OnUnitJam;

        }

        _hookedStations.Clear();
        UnhookMissileWarning();
        _subscribedAircraft = null;
    }

    private IEnumerator WaitAndSubscribeInstanceEvents()
    {
        // Wait until the singleton instance is non-null
        while (DynamicMap.i == null)
            yield return null;

        var dm = DynamicMap.i;
      //  dm.onMapGenerated += HandleMapGenerated;
        dm.onMapMaximized += HandleMapMaximized;
        dm.onMapMinimized += HandleMapMinimized;
       // dm.onAllDeselected += HandleAllDeselected;
        //dm.onUnitSelected += HandleUnitSelected;
        //dm.onUnitDeselected += HandleUnitDeselected;

        Debug.Log("[DebugDynamicMapLogger] Subscribed to DynamicMap instance events");
    }

    //not used?
    private void HookMissileWarning(MissileWarning mw)
    {
        if (mw == null) return;

        _subscribedMW = mw;
        mw.onMissileWarning += OnMissileWarning;
        mw.offMissileWarning += OffMissileWarning;

        Debug.Log($"[DebugEventLogger]    ↳ Hooked MissileWarning component '{mw.name}'");
    }
    //not used?
    private void UnhookMissileWarning()
    {
        if (_subscribedMW == null) return;

        _subscribedMW.onMissileWarning -= OnMissileWarning;
        _subscribedMW.offMissileWarning -= OffMissileWarning;

        Debug.Log($"[DebugEventLogger]    ↳ Unhooked MissileWarning");
        _subscribedMW = null;
    }

    // select target
    [HarmonyPatch(typeof(Aircraft), nameof(Aircraft.SetStationTargets), new Type[] { typeof(byte), typeof(int[]) })]
    static class Aircraft_SetStationTargets_Patch
    {
        static void Prefix(Aircraft __instance, byte stationIndex, int[] targetIDs)
        {
            var names = new List<string>();
            if (targetIDs != null)
            {
                foreach (var id in targetIDs)
                {
                    if (global::UnitRegistry.TryGetUnit(id, out var unit) && unit != null)
                        names.Add(unit.unitName);
                    else
                        names.Add($"#{id}");
                }
            }

            Debug.Log(
                $"[DebugEventLogger] SetStationTargets on '{__instance.unitName}' " +
                $"→ station={stationIndex}, targets=[{string.Join(", ", names)}]"
            );
        }
    }
